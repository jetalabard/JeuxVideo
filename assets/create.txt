CREATE TABLE ABOUT (ID INTEGER PRIMARY KEY  NOT NULL  UNIQUE , IDEA VARCHAR, DATA VARCHAR,URL_DATA VARCHAR,URL_FACEBOOK VARCHAR, MESS_SITE VARCHAR,URL_SITE VARCHAR,WHERE_DATA VARCHAR);
CREATE TABLE CONSOLE (ID INTEGER PRIMARY KEY  NOT NULL  UNIQUE , NAME VARCHAR, IMAGE VARCHAR);
CREATE TABLE FLUX (ID INTEGER PRIMARY KEY  NOT NULL  UNIQUE , CONSOLE INTEGER,  URL VARCHAR,CHOICE INTEGER);
CREATE TABLE CHOICE (ID INTEGER PRIMARY KEY  NOT NULL  UNIQUE , TITLE INTEGER, INTITULE VARCHAR,SUBTITLE VARCHAR,IMAGE VARCHAR);
CREATE TABLE MAJ (ID INTEGER PRIMARY KEY  NOT NULL  UNIQUE , TEXT VARCHAR, BUTTON VARCHAR);
